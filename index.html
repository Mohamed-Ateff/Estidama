نمو استراتيجي لمستقبـل مُستـدام

نـــظــرة عـــامة

مقدمة

تتبنى شركة مهارة للموارد البشرية مبادئ الاستدامة في جميع جوانب عملها، انطلاقاً من إيمان الشركة بأهمية تحقيق الاستدامة البيئية والاجتماعية وفي مجال الحوكمة. وتسعى الشركة في جميع القطاعات الي تعمل فيها إلى تحقيق التنمية الشاملة والتوازن بن الجوانب الاقتصادية والبيئية والاجتماعية.

رسـالـة مجلس الإدارة

رئيس مجلس الإدارة

سليمان بن عبد العزيز الماجد

السادة الأفاضل المساهمون وأصحاب المصلحة في شركة مهارة للموارد البشرية،،

أود أن أشكركم جميعاً على دعمكم المستمر للشركة، وإنه لمن دواعي سروري أن أقدم لكم تقرير الاستدامة للشركة لعام 2023 م، ذلك العام الذي يُعد بارزاً للشركة فيما بذلته من جهود كبرة في مجال الاستدامة . يُعد التقرير ركزة أساسية تواصل الشركة من خلالها تعزيز أولوياتها واستعراض تجاربها الرائدة نحو ترسيخ مبادئ الاستدامة وتقييم مدى نضج معاير الحوكمة البيئية والاجتماعية وحوكمة الشركات (ESG) من خلال وضع أجندة محددة المسارات وخارطة طريق لدعم مستقبل مُستدام.

"استراتيجية شاملة تحقق الاستدامة في الأعمال"

تأسســت "مهــارة" عــى مفهــوم الاســتدامة، ومنــذ تأسيســها ونحــن نعمــل عــى تقديــم الخدمــات المتوافقــة مــع معايــر الاســتدامة وتطويرهــا باســتمرار. كمــا انتهجــت الشركــة اســتراتيجية جيــدة وشــاملة لتحقيــق الاســتدامة فــي أعمالهــا. نجحنــا فــي "مهــارة"خــلال عــام 2023م، فــي تحقيــق تقــدم ملمــوس فــي مجــال الاســتدامة، والعمــل عــىوضــع خطــط اســتراتيجية فــي نمــوذج العمــل تهــدف إلى تقليــل الأثــر البيــي وتعزيــز المســؤولية الاجتماعيــة وتحقيــق المســاواة بــن الجنســن.كمــا واصلــت الشركــة العمــل عــى المســار الصحيــح لتحقيــق أهــداف وغايــات والتزامــات اســتراتيجية الشركــة الجديــدة تحــت مُســمى "اســتراتيجية نمــو مهارة ...نمو استراتيجي لمستقبل مُستدام."

تســعى "مهــارة" دومــاً إلى حمايــة حقــوق المســاهمن والحفــاظ عـلـى القيمــة الاســتثمارية وعـلـى الحــد مــن تضــارب المصالــح مــن خــلال الالــتزام بتطبيــق جميــع المعايــر وتعزيــز دور المســاهمن فــي المشــاركة فــي عمليــات اتخــاذ القــرارات الرئيسيةالمتعلقة بكل ما يرتبط باستثماراتهم.

نفخــر فــي "مهــارة"بالتقــدم الــذي أحرزنــاه عــى صعيــد تحفــز التنميــة المُســتدامة والتغيــر، ليــس عــى مســتوى الشركــة فحســب، بــل عــى المســتوى المجتمعــي، وذلـك وفقـاً لاسـتراتيجيتنا ورؤيتنـا الواضحـة أن نكـون أحـد رواد التنميـة المُسـتدامة والحوكمـة والممارسـات المسـؤولة لإحـداث تغيـر إيجابـي، وهـو مـا ينعكـس عـى عملائنا وموظفينا.

كمـا أننـا فـي "مهـارة"ننتهـج مسـاراً اسـتراتيجياً لإدمـاج معايـرالاسـتدامة الدوليـة وكيفيــة تطبيقهــا بمــا يضمــن تقليــل حجــم المخاطــر وتعزيــز قاعــدة أعمالنــا وأنشــطتنا القطاعيــة وتحســن مســتوى رضــا العمــلاء لتحقيــق أعــى مســتويات الخدمــة.كمــا نســعى دائمــاً إلى تطويــر اســتراتيجيتنا لتعزيــز هــذا الجانــب المهــم للشركة.

تتواصـل جهـود "مهـارة"فـي قيـادة النمـو المُسـتدام فـي جميـع الـشركات التابعـة لهــا والحفــاظ عــىقيمهــا الأساســية المتعلقــة بالابتــكار والتمكــن والعزيمــة، وســتثبت اســتراتيجيتنا للحوكمــة البيئيــة والاجتماعيــة وحوكمــة الــشركات )ESG(أهميـة مـا نقـوم بـه فـيإحـداث التغيـر المطلـوب لتحقيـق رؤيتنـا أن تصبـح شركـة مهــارة رائــدة فــي دعــم التنميــةالمُســتدامة فــي جميــع عملياتهــا داخــل وخــارج المملكة.

"نلتزم بتطبيق حوكمةقوية من خلال مجلس إدارةمسؤول"

تلــتزم "مهــارة"بتطبيــق حوكمــة قويــةمــن خــلال أداء مجلــس الإدارةالخاضــع للمسـاءلة والبيانـات وأمـن المعلومـات والإفصاحـات الـي تتسـم بالشـفافية ونهـج التحكــم فــي المخاطــر وإدارتهــا للحوكمــة البيئيــة والاجتماعيــة وحوكمــة الـشركات بالإضافــة إلى الممارســات الأخلاقيــةعــى مســتوى عمليــات الشركــة.

تعتمــد الشركــةسياســة حمايــة واحــترام حقــوق المســاهمن الــيتقــوم عــى مبـادئ الشـفافية والمسـاواة، بمـا فيهـا الالـتزام بالقوانـن والضوابـط والتعليمـات كافـة الـي تخضـع لهـا بشـأن حمايـةحقوقهـم بالإضافـة إلى تبـي سياسـة حمايـة حقوق الأطرافأصحاب المصالح.

"التحول الرقمي والتفوق يُعزز دورنا في خدمة العملاء"

ممــا لا شــك فيــه أن اســتراتيجيتنا فــي التحــول الرقمــي أظهــرت اســتجابة سريعــة تجــاه النمــو المتصاعــدلحلــول وخدمــات الشركــة الــي مــنشــأنها تطويــر خدمــة متطلبــات العمــلاء والمتماشــية مــع التحــول الشــامل الهــادف لرقمنــة خدمــات الشركــة، وفقــاً لاســتراتيجية مُمنهجــة، ولا ســيما مــع الطفــرة الــي يشــهدها التحول الرقمي في المملكة.

تسـعى "مهـارة" فـي نمـوذج أعمالهـا أن تكـون ضمـن أوائـل الـشركات الـي تعتمـد التحــول الرقمــي فــي أنشــطتها؛ إيمانــاً منهــا بأنــه نقطــة ارتــكاز قويــة للتوســع المحــي والإقليمــي. فــي هــذا الإطــار، تعمــل الشركــة دومــاً عــى تطويــر البنيــة التحتيــة الرقميــة واعتمــاد التكنولوجيــا الحديثــة فــي نمــوذج الأعمــال للتيســر عــى العملاء وتحسن تجربة العميل وتطويرها بشكل مستمر.

"القيادة والشمول في دعم مواردنا البشرية"

نعمــل فــي "مهــارة"عــى تهيئــة بيئــة عمــل تتمــز بالشــمول والتفاعــل لموظفينــا كونهــم المُحــرك الأســاسيللنجــاح المتواصــل الــذي تشــهده الشركــة مــن خــلال تبــي مجموعــة متنوعــةمــن المبــادرات والبرامــج التدريبيــة بالتعــاون مــع أعــرق المؤسســات العالميــة وأبــرز الخـبراء المتخصصــن فــي التدريــب لتعزيــز وتطويــر أهــم المسـارات الوظيفيـة وسُـبل التطـور المهـي وسـط بيئـة عمـل اسـتثنائية، ليصـل عدد الموظفــن الحاصلــن عــى برامــج تدريبيــة متخصصــة إلى 392 موظفــاً وموظفــة بواقـع 365 موظفـاًوموظفـة حصلـوا عـى دورات داخليـة،و27موظفـاً وموظفـة حصلــوا عــى دورات خارجيــة.

وتحـرص "مهـارة"عـى توفـر مهـاموظيفيـة متكافئـة ومتسـاوية لجميـع كوادرهـا الوطنيــة مــن الرجــال والنســاء؛ لضمــان تحقيــق مبــدأ المســاواة، وفيمــا يتعلــق بتمكــن المــرأة فــي المناصــب القياديــة، وصلــت نســبة العامــلات فــي الشركــة إلى 28.5 %، فيمــا ارتفعــت نســبةالســعودة عــى جميــعالمســتويات الوظيفيــة إلى 66.5 %، مــع تحقيــق نســبة ســعودةبلغــت 78.57 % عــى مســتوى الإدارة العليا.

"مسؤولية اجتماعية وبيئية تتفق مع المعايير الدولية"

لعــل أبــرز مــايمــز برنامــج المســؤولية الاجتماعيــة فــي "مهــارة" هــو التنــوع فــي المجــالات الــي تتناولهــا ودعمهــابصــورة واضحــة مــن خـلال مســاهمات الشركــة المختلفــة والأعمــال التطوعيــة والخريــة الــي ترمــي مــن خلالهــا إلى ترســيخ مبــدأ ضرورة المشــاركة المجتمعيــة للقطــاع الخــاص لتحقيــق الاســتدامة فــي مختلــف مجالات الأعمال.

وقـد عكسـت اسـتراتيجية المسـؤولية الاجتماعيـة فـي عـام 2023م الـتزام "مهـارة" تجــاه وضــع معايــر خاصــة تســتهدف تحقيــق تواصــل مســتمر مــع عملائهــا وموظفيهــا وغرهــم، ليؤثــر إيجابيــاً وبصــورة هادفــة عــى مســرة الشركــة وتطويرها لتحقيق نمو مُستدام.

تتضمــن جهودنــاأيضــاً الممارســات المُســتدامة فــي مجــال البيئــة،والــي تؤثــر فــي المجتمــع وتتأثــر بســلوكياته. لذلــك، نهتــم فــي "مهــارة"بقيــاس أثــر ممارســتنا العمليــة فــي مختلــف القطاعــات والحــدمــن الآثــارالســلبية للانبعاثــات الكربونيــة والاســتهلاك الزائــد للطاقــة والميــاه وتوليــد النفايــات الــي تــر بالبيئــة والمجتمــع، وقــد تــم وضــعخطــط واضحــة فــي "مهــارة" للمســاهمة فــي معالجــة كلهــذه الظواهر.

رحلتنـا الـي نخوضهـا فـي "مهـارة"ليسـت إلا البدايـة، فنحـن عازمـون عـى اسـتكمال المســرة معكــم،ونُــدرك أن تحقيــقأهــداف الممارســات البيئيــة والاجتماعيــة والحوكمــة لــه الــتزام مســتمر يتطلــب عزيمــة وصــبراً وانتباهــاً لا يــكلّ بمــا يضمــن مواصلــة التقــدموالإنجــاز، حيــث نتطلــعلمناقشــة تفاصيــل هــذا الرحلــة معكــم فــي تقريــر الاســتدامة،ونســعد بالعمــل معكــم عــى إيجــاد حلــول أكــر اســتدامة ومسؤولية تجعل النجاح حليفنا في "مهارة."

ختامـاً.. شـكراً لـكل مَـنْ سـاهم فـي هـذه الرحلـة مـن دعـم ومشـاركة، والـي مـن دونهــا لــن يتســى لنــا بنــاءالمســتقبل المُــشرق والمُســتدام الــذي نســعى إليــه، متطلعين -بمشيئة الله- إلى إنجازات ونجاحات جديدة.
